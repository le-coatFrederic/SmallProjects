1   package webcrawler.application.crawler;
2   
3   import java.util.ArrayList;
4   import java.util.HashMap;
5   
6   import webcrawler.domain.entities.Site;
7   import webcrawler.domain.usecase.crawler.ICrawlerURLGet;
8   import webcrawler.domain.values.URL;
9   
10  public class URLInArray implements ICrawlerURLGet {
11      private ArrayList<URL> urls = new ArrayList<>();
12      private HashMap<String, URL> urlsVisited = new HashMap<>();
13  
14      public void addUrl(URL urlToAdd) {
15          if (urlToAdd == null) {
16              throw new IllegalArgumentException();
17          }
18  
19          if (urlsVisited.containsKey(urlToAdd.getUrl())) {
20              return;
21          }
22  
23          urls.add(urlToAdd);
24      }
25  
26      public ArrayList<URL> getUrls() {
27          return urls;
28      }
29  
30      public HashMap<String, URL> getVisitedUrls() {
31          return urlsVisited;
32      }
33  
34      @Override
35      public synchronized URL getUrl() {
36          if (urls.isEmpty()) {
37              throw new NullPointerException();
38          }
39  
40          URL url = urls.get(0);
41          urls.remove(0);
42          return url;
43      }
44  
45      @Override
46      public ArrayList<URL> saveUrlOfSite(Site site) {
47          if (site == null) {
48              throw new IllegalArgumentException();
49          }
50  
51          if (urlsVisited.containsKey(site.getUrl().getUrl())) {
52              throw new IllegalArgumentException();
53          }
54  
55          urls.add(site.getUrl());
56          return this.urls;
57      }
58      
59  }
