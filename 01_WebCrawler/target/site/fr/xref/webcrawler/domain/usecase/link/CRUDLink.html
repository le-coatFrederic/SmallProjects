6   import webcrawler.domain.entities.Link;

9   public class CRUDLink implements ILinkData{
10  	private HashMap<Integer, Link> links;
11  	private ILinkDbInput dbInput;
12  	private ILinkDbOutput dbOutput;
13  
14  	public CRUDLink(ILinkDbInput dbInput, ILinkDbOutput dbOutput) {
15  		if (dbInput == null ||dbOutput == null) {
16  			throw new IllegalArgumentException();
17  		}
18  
19  		links = new HashMap<Integer, Link>();
20  		this.dbInput = dbInput;
21  		this.dbOutput = dbOutput;
22  	}
23  
24  	public ILinkDbInput getDbInput() {
25  		return this.dbInput;
26  	}
27  
28  	public ILinkDbOutput geDbOutput() {
29  		return this.dbOutput;
30  	}
31  
32  	public CRUDLink setDbInput(ILinkDbInput dbInput) {
33  		if (dbInput == null) {
34  			throw new IllegalArgumentException();
35  		}
36  
37  		this.dbInput = dbInput;
38  		return this;
39  	}
40  
41  	public CRUDLink setDbOutput(ILinkDbOutput dbOutput) {
42  		if (dbOutput == null) {
43  			throw new IllegalArgumentException();
44  		}
45  
46  		this.dbOutput = dbOutput;
47  		return this;
48  	}
49  
50  	@Override
51  	public Link createLink(int id, URL site1, URL site2) {
52  		if (site1 == null || site2 == null) {
53  			throw new NullPointerException();
54  		}
55  		
56  		if (links.containsKey(id)) {
57  			throw new IllegalArgumentException();
58  		}
59  		
60  		Link newLink = new Link(id, site1, site2);
61  		links.put(id, newLink);
62  		
63  		return newLink;
64  	}
65  
66  	@Override
67  	public Link updateLink(int id, URL site1, URL site2) {
68  		if (links.containsKey(id)) {
69  			throw new IllegalArgumentException();
70  		}
71  		
72  		Link updatedLink = links.get(id);
73  		
74  		if (site1 != null) {
75  			updatedLink.setSite1(site1);
76  		}
77  		
78  		if (site2 != null) {
79  			updatedLink.setSite2(site2);
80  		}
81  		
82  		return updatedLink;
83  	}
84  
85  	@Override
86  	public Link detailLink(int id) {
87  		if (!links.containsKey(id)) {
88  			throw new IllegalArgumentException();
89  		}
90  		
91  		return links.get(id);
92  	}
